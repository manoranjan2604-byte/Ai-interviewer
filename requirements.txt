# ---- LLM providers (config.LLM_PROVIDER: cerebras | groq) ----
# Both Cerebras and Groq expose OpenAI-compatible endpoints, so the
# openai SDK (pointed at a different base_url) covers both.
openai==1.42.0

# ---- Speech-to-text (config.STT_PROVIDER: whisper | groq) ----
# groq (default): uses the same openai SDK above, no extra package needed,
# and needs no ffmpeg/Docker -- this is what makes Render's plain Python
# runtime (no system-package installs) able to run STT at all.
#
# whisper: local, no API key, needs ffmpeg on PATH -- but pulls in PyTorch
# (a very heavy install) and needs real CPU/RAM headroom neither of which
# fit Render's free/cheap tiers well. Uncomment only if you're switching
# STT_PROVIDER back to "whisper" (e.g. for local dev without a GROQ_API_KEY):
# openai-whisper

# ---- Text-to-speech (config.TTS_PROVIDER: local | edge) ----
# local: pyttsx3, offline, no API key, robotic voice
pyttsx3
# edge: free, unlimited, no API key, better voice quality
edge-tts

# ---- Durability layer (optional) ----
# redis: only used if REDIS_URL is set in .env — mirrors session status
# and JSON reports so they survive a restart. See config.py for details.
redis==5.0.8
